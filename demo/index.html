typedef bool Compare(int a, int b); 

bool insertion_sort(Array(int) in, Compare* compare) {

    if (in.count < 2) return false; // can't sort
    
    for (u64 i = 1; i < in.count; i++) {

        int key = in.data[i];
        
        // move stuff up for the insertion
        u64 j = i;
        while (j > 0 && compare(key, in.data[j - 1])) {
            in.data[j] = in.data[j - 1];
            j--;
        }

        in.data[j] = key;
        
        #if DEBUG == true
            printf("Inserted [%d] to %llu\n", key, j);
        #endif
    }

    return true;
}
